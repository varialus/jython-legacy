from org.eclipse.core.resources import IMarker as MRK

jy = env.got
out(jy)

markers = jy.resource.findMarkers(MRK.TASK,1,jy.resource.DEPTH_INFINITE)

out("tasks #: %d" % len(markers))

kinds = {}

for m in markers:
   msg = m.getAttribute("message")
   kind = msg.split()[0].upper()
   kinds[kind] = kinds.get(kind,0)+1

out(kinds)
